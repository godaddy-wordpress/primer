250: /**
251:  * Display very simple breadcrumbs.
252:  *
253:  * Adapted from Christoph Weil's Really Simple Breadcrumb plugin.
254:  *
255:  * @global WP_Post $post
256:  * @link   https://wordpress.org/plugins/really-simple-breadcrumb/
257:  * @since  1.0.0
258:  */
259: function primer_breadcrumbs() {
260: 
261:     global $post;
262: 
263:     $separator = ' <span class="sep"></span> ';
264: 
265:     echo '<div class="breadcrumbs">';
266: 
267:     if ( ! is_front_page() ) {
268: 
269:         printf(
270:             '<a href="%s">%s</a>%s',
271:             esc_url( home_url( '/' ) ),
272:             esc_html( get_bloginfo( 'name' ) ),
273:             $separator // xss ok
274:         );
275: 
276:         if ( 'page' === get_option( 'show_on_front' ) ) {
277: 
278:             printf(
279:                 '<a href="%s">%s</a>%s',
280:                 esc_url( primer_get_posts_url() ),
281:                 esc_html__( 'Blog', 'primer' ),
282:                 $separator // xss ok
283:             );
284: 
285:         }
286: 
287:         if ( is_category() || is_single() ) {
288: 
289:             the_category( ', ' );
290: 
291:             if ( is_single() ) {
292: 
293:                 echo $separator; // xss ok
294: 
295:                 the_title();
296: 
297:             }
298: 
299:         } elseif ( is_page() && $post->post_parent ) {
300: 
301:             $home = get_post( get_option( 'page_on_front' ) );
302: 
303:             for ( $i = count( $post->ancestors )-1; $i >= 0; $i-- ) {
304: 
305:                 if ( ( $home->ID ) != ( $post->ancestors[$i] ) ) {
306: 
307:                     echo '<a href="' . get_permalink( $post->ancestors[$i] ) . '">' . get_the_title( $post->ancestors[$i] ) . '</a>' . $separator;
308: 
309:                 }
310:             }
311: 
312:             the_title();
313: 
314:         } elseif ( is_page() ) {
315: 
316:             the_title();
317: 
318:         } elseif ( is_404() ) {
319: 
320:             echo '404';
321: 
322:         }
323: 
324:     } else {
325: 
326:         bloginfo( 'name' );
327: 
328:     }
329: 
330:     echo '</div>';
331: 
332: }
