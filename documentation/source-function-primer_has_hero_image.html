 18: function primer_get_the_page_title() {
 19: 
 20:     $title = '';
 21: 
 22:     switch ( true ) {
 23: 
 24:         case is_front_page() :
 25: 
 26:             $title = get_the_title( get_option( 'page_on_front' ) );
 27: 
 28:             break;
 29: 
 30:         case is_home() :
 31: 
 32:             $title = get_the_title( get_option( 'page_for_posts' ) );
 33: 
 34:             break;
 35: 
 36:         case is_archive() :
 37: 
 38:             $title = get_the_archive_title();
 39: 
 40:             break;
 41: 
 42:         case is_search() :
 43: 
 44:             $title = sprintf(
 45:                 esc_html_x( 'Search Results for: %s', 'search term', 'primer' ),
 46:                 sprintf(
 47:                     '<span>%s</span>',
 48:                     get_search_query()
 49:                 )
 50:             );
 51: 
 52:             break;
 53: 
 54:         case is_404() :
 55: 
 56:             $title = esc_html__( '404 Page Not Found', 'primer' );
 57: 
 58:             break;
 59: 
 60:         case is_page() :
 61: 
 62:             $title = get_the_title();
 63: 
 64:             break;
 65: 
 66:         case ( ( $post = get_queried_object() ) && ! is_post_type_hierarchical( get_post_type( $post ) ) ) :
 67: 
 68:             $show_on_front  = get_option( 'show_on_front' );
 69:             $page_for_posts = get_option( 'page_for_posts' );
 70: 
 71:             if ( 'post' === $post->post_type && 'posts' !== $show_on_front && ! empty( $page_for_posts ) ) {
 72: 
 73:                 $title = get_the_title( $page_for_posts );
 74: 
 75:                 break;
 76: 
 77:             }
 78: 
 79:             $labels = get_post_type_labels( get_post_type_object( $post->post_type ) );
 80: 
 81:             $title = isset( $labels->name ) ? $labels->name : false;
 82: 
 83:             break;
 84: 
 85:     }

 88:      * Filter the page title.
 89:      *
 90:      * @since 1.0.0
 91:      *
 92:      * @var string

518: /**
519:  * Convert a 3- or 6-digit hexadecimal color to an associative RGB array.
520:  *
521:  * @since  1.0.0
522:  *
523:  * @param  string $color
524:  *
525:  * @return array Returns an array of Red, Green and Blue values to use in CSS styles.
526:  */
527: function primer_hex2rgb( $color ) {
528: 
529:     $color = trim( $color, '#' );
530: 
531:     switch ( strlen( $color ) ) {
532: 
533:         case 3 :
534: 
535:             $r = hexdec( substr( $color, 0, 1 ) . substr( $color, 0, 1 ) );
536:             $g = hexdec( substr( $color, 1, 1 ) . substr( $color, 1, 1 ) );
537:             $b = hexdec( substr( $color, 2, 1 ) . substr( $color, 2, 1 ) );
538: 
539:             break;
540: 
541:         case 6 :
542: 
543:             $r = hexdec( substr( $color, 0, 2 ) );
544:             $g = hexdec( substr( $color, 2, 2 ) );
545:             $b = hexdec( substr( $color, 4, 2 ) );
546: 
547:             break;
548: 
549:         default :
550: 
551:             return array();
552: 
553:     }
554: 
555:     return array( 'red' => $r, 'green' => $g, 'blue' => $b );
556: 
557: }

559: /**
560:  * Recursively replace elements from passed arrays into the first array (safe for PHP 5.2).
561:  *
562:  * @author Frankie Jarrett <fjarrett@gmail.com>
563:  * @link   http://php.net/manual/en/function.array-replace-recursive.php
564:  * @since  1.0.0
565:  *
566:  * @param  array $array1
567:  * @param  array $array2
568:  *
569:  * @return array Returns a merged array.
570:  */
571: function primer_array_replace_recursive( array $array1, array $array2 ) {
572: 
573:     if ( function_exists( 'array_replace_recursive' ) ) {
574: 
575:         return call_user_func_array( 'array_replace_recursive', func_get_args() );
576: 
577:     }
578: 
579:     $result = array();
580: 
581:     for ( $i = 0, $total = func_num_args(); $i < $total; $i++ ) {
582: 
583:         $_array = func_get_arg( $i );
584: 
585:         foreach ( $_array as $key => &$value ) {
586: 
587:             if ( is_array( $value ) && isset( $result[ $key ] ) && is_array( $result[ $key ] ) ) {
588: 
589:                 $result[ $key ] = call_user_func( __FUNCTION__, $result[ $key ], $value );
590: 
591:                 continue;
592: 
593:             }
594: 
595:             $is_assoc = ( array_keys( $_array ) !== range( 0, count( $_array ) - 1 ) );
596: 
597:             if ( ! $is_assoc && ! in_array( $value, $merged, true ) ) {
598: 
599:                 $result[] = $value;
600: 
601:                 continue;
602: 
603:             }
604: 
605:             $result[ $key ] = $value;
606: 
607:         }
608: 
609:     }
610: 
611:     return (array) $result;
612: 
613: }
