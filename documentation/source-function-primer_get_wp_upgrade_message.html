  3:  * WordPress backwards compatibility.
  4:  *
  5:  * Prevents Primer from running on older WordPress versions since
  6:  * this theme is not meant to be backward compatible beyond two
  7:  * major versions and relies on many newer functions and markup.
  8:  *
  9:  * @package    Compatibility
 10:  * @subpackage WordPress
 11:  * @category   Core
 12:  * @author     GoDaddy
 13:  * @since      1.0.0
 14:  */
 15: 
 16: /**
 17:  * Switch to the default theme immediately.
 18:  *
 19:  * @action admin_notices
 20:  * @action after_setup_theme
 21:  * @uses   [switch_theme](https://codex.wordpress.org/Function_Reference/switch_theme) To switch WordPress themes.
 22:  *
 23:  * @since  1.0.0
 24:  */
 25: function primer_switch_theme() {
 26: 
 27:     switch_theme( WP_DEFAULT_THEME, WP_DEFAULT_THEME );
 28: 
 29:     unset( $_GET['activated'] ); // input var ok.
 30: 
 31:     add_action( 'admin_notices', 'primer_upgrade_notice' );
 32: 
 33: }
 34: add_action( 'after_setup_theme', 'primer_switch_theme', 1 );
 35: 
 36: /**
 37:  * Return the required WordPress version upgrade message.
 38:  *
 39:  * @action after_setup_theme
 40:  * @uses   [get_bloginfo](https://codex.wordpress.org/Function_Reference/get_bloginfo) To retreive the WordPress version.
 41:  *
 42:  * @since  1.0.0
 43:  */
 44: function primer_get_wp_upgrade_message() {
 45: 
 46:     /**
 47:      * Filter the required WordPress version upgrade message.
 48:      *
 49:      * @since 1.0.0
 50:      *
 51:      * @var string
 52:      */
 53:     return (string) apply_filters( 'primer_required_wp_version_message',
 54:         sprintf(
 55:             esc_html__( 'Primer requires at least WordPress version %1$s. You are running version %2$s. Please upgrade and try again.', 'primer' ),
 56:             PRIMER_MIN_WP_VERSION,
 57:             get_bloginfo( 'version' )
 58:         )
 59:     );
 60: 
 61: }
 62: 
 63: /**
 64:  * Adds a message for unsuccessful theme switch.
 65:  *
 66:  * Prints an update nag after an unsuccessful attempt to
 67:  * activate Primer on older WordPress versions.
 68:  *
 69:  * @since 1.0.0
 70:  */
 71: function primer_upgrade_notice() {
 72: 
 73:     printf( '<div class="error"><p>%s</p></div>', esc_html( primer_get_wp_upgrade_message() ) );
 74: 
 75: }
 76: 
 77: /**
 78:  * Prevents the Customizer from being loaded on older WordPress versions.
 79:  *
 80:  * @action load-customize.php
 81:  * @since  1.0.0
 82:  */
 83: function primer_customize() {
 84: 
 85:     wp_die( esc_html( primer_get_wp_upgrade_message() ), '', array( 'back_link' => true ) );
 86: 
 87: }
 88: add_action( 'load-customize.php', 'primer_customize' );

 90: /**
 91:  * Prevents the Theme Preview from being loaded on older WordPress versions.

 93:  * @action template_redirect
 94:  * @since  1.0.0
 95:  */
 96: function primer_preview() {
 97: 
 98:     if ( isset( $_GET['preview'] ) ) { // input var ok.
 99: 
100:         wp_die( esc_html( primer_get_wp_upgrade_message() ) );
101: 
102:     }
103: 
104: }
